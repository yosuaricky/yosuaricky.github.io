Cleaning the data and assign into new variable

penguin_data <- penguins %>% 
  drop_na()

Create plot for the penguin

penguin_data %>% 
  ggplot() +
  geom_point(mapping = aes(x = flipper_length_mm, y = body_mass_g, color = species))

Create plot for each penguin species

penguin_data %>% 
  ggplot() +
  geom_point(mapping = aes(x = flipper_length_mm, y = body_mass_g, color = species)) +
  facet_wrap(~species)
